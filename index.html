import telebot import json from telebot.types import ReplyKeyboardMarkup, KeyboardButton, WebAppInfo BOT_TOKEN = "YOUR_BOT_TOKEN_HERE" # Yahan apna asli token dalein bot = telebot.TeleBot(BOT_TOKEN) # Jab aap HTML file ko host kar lenge, toh uska link yahan dalna hoga WEB_APP_URL = "https://your-hosted-web-app-url.com" @bot.message_handler(commands=['start']) def start_message(message): markup = ReplyKeyboardMarkup(resize_keyboard=True) web_app_btn = KeyboardButton(text="🛍️ Store Open Karein", web_app=WebAppInfo(url=WEB_APP_URL)) markup.add(web_app_btn) bot.send_message( message.chat.id, "Welcome! Niche diye gaye button par click karke hamare store se shopping karein:", reply_markup=markup ) # Mini App se order data receive karna @bot.message_handler(content_types=['web_app_data']) def web_app_data(message): try: # Received data ko parse karna data = json.loads(message.web_app_data.data) items = data.get("items") total = data.get("total") response_text = ( f"✅ **Naya Order Mila Hai!**\n\n" f"🛒 **Items:** {items}\n" f"💰 **Total Amount:** ₹{total}\n\n" f"Ab kripya apna Delivery Address aur Mobile Number type karke bhejein." ) bot.send_message(message.chat.id, response_text, parse_mode="Markdown") except Exception as e: bot.send_message(message.chat.id, "Order receive karne me kuch dikkat aayi.") print("Bot successfully chal raha hai...") bot.polling()
